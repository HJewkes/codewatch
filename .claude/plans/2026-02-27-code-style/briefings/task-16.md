# Task 16: ESLint + Ruff + Misc Exports

## Architectural Context

This task implements four additional exporters in the `profile` package and a unified `exportProfile` dispatch function. The ESLint exporter reuses the checker's config generator (Task 14) but serializes to a standalone `eslint.config.js` file. The Ruff exporter writes `ruff.toml`. The Markdown exporter generates a human-readable style guide. The EditorConfig exporter maps formatting rules to `.editorconfig`. The export index provides a single `exportProfile(profile, format, outputPath)` entry point consumed by the `export` command in Task 18.

import { generateEslintConfig } from "@code-style/checker";
import type { StyleProfile } from "../schema/profile.js";
import type { GeneratedFile } from "./skill.js";

function detectRequiredPlugins(configStr: string): string[] {
  const plugins: string[] = [];
  if (configStr.includes("@typescript-eslint"))
    plugins.push("@typescript-eslint/eslint-plugin", "@typescript-eslint/parser");
  if (configStr.includes("perfectionist"))
    plugins.push("eslint-plugin-perfectionist");
  if (configStr.includes("unicorn"))
    plugins.push("eslint-plugin-unicorn");
  if (configStr.includes("import/"))
    plugins.push("eslint-plugin-import");
  if (configStr.includes("jsdoc"))
    plugins.push("eslint-plugin-jsdoc");
  return plugins;
}

function generateImports(plugins: string[]): string {
  const imports: string[] = [];
  if (plugins.some((p) => p.includes("@typescript-eslint"))) {
    imports.push(`import tseslint from "@typescript-eslint/eslint-plugin";`);
    imports.push(`import tsparser from "@typescript-eslint/parser";`);
  }
  if (plugins.some((p) => p.includes("perfectionist"))) {
    imports.push(`import perfectionist from "eslint-plugin-perfectionist";`);
  }
  if (plugins.some((p) => p.includes("unicorn"))) {
    imports.push(`import unicorn from "eslint-plugin-unicorn";`);
  }
  if (plugins.some((p) => p.includes("jsdoc"))) {
    imports.push(`import jsdoc from "eslint-plugin-jsdoc";`);
  }
  return imports.join("\n");
}

export function generateEslintExport(profile: StyleProfile): GeneratedFile {
  const config = generateEslintConfig(profile);
  const configJson = JSON.stringify(config, null, 2);
  const plugins = detectRequiredPlugins(configJson);
  const imports = generateImports(plugins);

  const header = [
    `// Generated by code-style for ${profile.author}`,
    `// Date: ${profile.generated}`,
    `// Do not edit manually — regenerate with: code-style export --format eslint`,
    "",
  ].join("\n");

  const content = [
    header,
    imports,
    "",
    `export default ${configJson};`,
    "",
  ].join("\n");

  return { path: "eslint.config.js", content };
}

### Step 4: Implement Ruff exporter

Create `/packages/profile/src/exporters/ruff.ts`:

```ts
import { generateRuffConfig } from "@code-style/checker";
import type { RuffConfig } from "@code-style/checker";
import type { StyleProfile } from "../schema/profile.js";
import type { GeneratedFile } from "./skill.js";

function toToml(config: RuffConfig): string {
  const lines: string[] = [];

  if (config["line-length"]) {
    lines.push(`line-length = ${config["line-length"]}`);
    lines.push("");
  }

  if (config.lint) {
    lines.push("[lint]");
    if (config.lint.select) {
      lines.push(`select = [${config.lint.select.map((s) => `"${s}"`).join(", ")}]`);
    }
    if (config.lint.ignore) {
      lines.push(`ignore = [${config.lint.ignore.map((s) => `"${s}"`).join(", ")}]`);
    }
    if (config.lint.fixable) {
      lines.push(`fixable = [${config.lint.fixable.map((s) => `"${s}"`).join(", ")}]`);
    }
    lines.push("");

    if (config.lint.mccabe) {
      lines.push("[lint.mccabe]");
      if (config.lint.mccabe["max-complexity"] !== undefined) {
        lines.push(`max-complexity = ${config.lint.mccabe["max-complexity"]}`);
      }
      lines.push("");
    }

    if (config.lint.pydocstyle) {
      lines.push("[lint.pydocstyle]");
      if (config.lint.pydocstyle.convention) {
        lines.push(`convention = "${config.lint.pydocstyle.convention}"`);
      }
      lines.push("");
    }

    if (config.lint.isort) {
      lines.push("[lint.isort]");
      if (config.lint.isort["section-order"]) {
        lines.push(
          `section-order = [${config.lint.isort["section-order"].map((s) => `"${s}"`).join(", ")}]`,
        );
      }
      lines.push("");
    }
  }

  return lines.join("\n");
}

export function generateRuffExport(profile: StyleProfile): GeneratedFile {
  const config = generateRuffConfig(profile);

  const header = [
    `# Generated by code-style for ${profile.author}`,
    `# Date: ${profile.generated}`,
    `# Do not edit manually — regenerate with: code-style export --format ruff`,
    "",
  ].join("\n");

  return {
    path: "ruff.toml",
    content: header + toToml(config),
  };
}
```

### Step 6: Implement Markdown exporter

Create `/packages/profile/src/exporters/markdown.ts`:

```ts
import type { StyleProfile } from "../schema/profile.js";
import { extractAllRules } from "./template-helpers.js";
import type { GeneratedFile } from "./skill.js";

function capitalize(str: string): string {
  return str.charAt(0).toUpperCase() + str.replace(/([A-Z])/g, " $1").slice(1);
}

export function generateMarkdownExport(profile: StyleProfile): GeneratedFile {
  const allRules = extractAllRules(profile);
  const byCategory = new Map<string, typeof allRules>();

  for (const rule of allRules) {
    const existing = byCategory.get(rule.category) ?? [];
    existing.push(rule);
    byCategory.set(rule.category, existing);
  }

  const lines: string[] = [];

  lines.push(`# ${profile.author}'s Code Style Guide`);
  lines.push("");
  lines.push(`> Generated by code-style on ${profile.generated}`);
  lines.push(`> Sources: ${profile.sources.join(", ") || "N/A"}`);
  lines.push("");

  for (const [category, rules] of byCategory) {
    lines.push(`## ${capitalize(category)}`);
    lines.push("");

    const sorted = rules.sort((a, b) => b.confidence - a.confidence);
    for (const rule of sorted) {
      const confPct = (rule.confidence * 100).toFixed(0);
      const value =
        typeof rule.convention === "string"
          ? rule.convention
          : JSON.stringify(rule.convention);

      lines.push(`### ${capitalize(rule.name)}`);
      lines.push("");
      lines.push(`**Convention**: \`${value}\` | **Confidence**: ${confPct}%${rule.stability ? ` | **Stability**: ${rule.stability}` : ""}`);
      lines.push("");

      if (rule.description) {
        lines.push(rule.description);
        lines.push("");
      }

      if (rule.examples && rule.examples.length > 0) {
        for (const example of rule.examples) {
          if (example.good) {
            lines.push("```typescript");
            lines.push(`// Preferred`);
            lines.push(example.good);
            lines.push("```");
            lines.push("");
          }
          if (example.bad) {
            lines.push("```typescript");
            lines.push(`// Avoid`);
            lines.push(example.bad);
            lines.push("```");
            lines.push("");
          }
        }
      }
    }
  }

  return {
    path: "style-guide.md",
    content: lines.join("\n"),
  };
}
```

import type { StyleProfile } from "../schema/profile.js";
import type { GeneratedFile } from "./skill.js";

function getConvention<T>(
  section: unknown,
  key: string,
): T | undefined {
  if (!section || typeof section !== "object") return undefined;
  const rule = (section as Record<string, unknown>)[key];
  if (!rule || typeof rule !== "object") return undefined;
  return (rule as { convention?: T }).convention;
}

export function generateEditorConfigExport(
  profile: StyleProfile,
): GeneratedFile {
  const formatting = profile.formatting;
  const lines: string[] = [];

  lines.push("# Generated by code-style");
  lines.push("root = true");
  lines.push("");
  lines.push("[*]");

  const indentStyle = getConvention<string>(formatting, "indentStyle");
  if (indentStyle) {
    const editorConfigValue = indentStyle === "spaces" ? "space" : indentStyle === "tabs" ? "tab" : indentStyle;
    lines.push(`indent_style = ${editorConfigValue}`);
  }

  const indentSize = getConvention<number>(formatting, "indentSize");
  if (indentSize !== undefined) {
    lines.push(`indent_size = ${indentSize}`);
  }

  const trailingNewline = getConvention<boolean>(formatting, "trailingNewline");
  if (trailingNewline !== undefined) {
    lines.push(`insert_final_newline = ${trailingNewline}`);
  }

  lines.push("end_of_line = lf");
  lines.push("charset = utf-8");
  lines.push("trim_trailing_whitespace = true");

  const lineLength = getConvention<number>(formatting, "lineLength");
  if (lineLength !== undefined) {
    lines.push(`max_line_length = ${lineLength}`);
  }

  lines.push("");

  return {
    path: ".editorconfig",
    content: lines.join("\n"),
  };
}

import type { StyleProfile } from "../schema/profile.js";
import type { GeneratedFile } from "./skill.js";
import { generateSkillFiles } from "./skill.js";
import { generateClaudeRules } from "./claude-rules.js";
import { generateHooksConfig } from "./hooks.js";
import { generateEslintExport } from "./eslint.js";
import { generateRuffExport } from "./ruff.js";
import { generateMarkdownExport } from "./markdown.js";
import { generateEditorConfigExport } from "./editorconfig.js";

export type ExportFormat =
  | "skill"
  | "claude-rules"
  | "hooks"
  | "eslint"
  | "ruff"
  | "markdown"
  | "editorconfig";

export const SUPPORTED_FORMATS: ExportFormat[] = [
  "skill",
  "claude-rules",
  "hooks",
  "eslint",
  "ruff",
  "markdown",
  "editorconfig",
];

export function exportProfile(
  profile: StyleProfile,
  format: ExportFormat,
): GeneratedFile[] {
  switch (format) {
    case "skill":
      return generateSkillFiles(profile);

    case "claude-rules":
      return generateClaudeRules(profile);

    case "hooks": {
      const config = generateHooksConfig(profile);
      return [
        {
          path: ".claude/settings.json",
          content: JSON.stringify(config, null, 2),
        },
      ];
    }

    case "eslint":
      return [generateEslintExport(profile)];

    case "ruff":
      return [generateRuffExport(profile)];

    case "markdown":
      return [generateMarkdownExport(profile)];

    case "editorconfig":
      return [generateEditorConfigExport(profile)];

    default:
      throw new Error(
        `Unsupported export format: "${format}". Supported formats: ${SUPPORTED_FORMATS.join(", ")}`,
      );
  }
}

## Anti-patterns

### Universal
1. **Do not install packages globally** -- all deps go in the workspace root or specific packages
2. **Do not skip the verify step** -- every file must compile before committing
3. **Do not create files outside the defined file ownership list**

### Task-specific
4. **Do not duplicate the ESLint/Ruff config generation logic** -- import and reuse `generateEslintConfig` / `generateRuffConfig` from `@code-style/checker`; only add serialization-to-file on top
5. **Do not write files to disk in the exporter functions** -- exporters return `GeneratedFile[]` with path and content; the CLI command handles actual file I/O
6. **Do not generate TOML with a TOML library** -- the ruff config structure is simple enough for string concatenation; adding a TOML dependency is unnecessary
7. **Do not include low-confidence rules in exports without indication** -- every rule in every export format should show its confidence level or only include rules above the info threshold
